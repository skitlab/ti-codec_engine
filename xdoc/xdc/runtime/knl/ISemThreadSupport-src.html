    17    /*! 
    18     *  ======== ISemThreadSupport ========
    19     *  Interface for OS specific back-end.
    20     *
    21     *  The {@link xdc.runtime.knl} package contains modules that provide typical
    22     *  OS services. These xdc.runtime.knl modules require proxies to be
    23     *  bound to an OS specific delegate. This specifies the interface to 
    24     *  be implemented by the OS specific delegate for 
    25     *  {@link xdc.runtime.knl#SemThread} module.
    26     * 
    27     *  This interface adds the create() function.
    28     */
    29    interface ISemThreadSupport inherits ISemaphore
    30    {
    31    
    32    instance:
    33    
    34       /*!
    35        *  ======== create ========
    36        *  Create a Semaphore object
    37        *
    38        *  This function creates a new Semaphore object which is initialized to
    39        *  count.
    40        *
    41        *  @param(count)    initial semaphore count
    42        */
    43        create(Int count);
    44    }
