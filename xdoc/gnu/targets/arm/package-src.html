    13    requires xdc.bld;
    14    requires gnu.targets;

    16    /*!
    17     *  ======== gnu.targets.arm ========
    18     *  GCC-based code generation targets for Arm
    19     *
    20     *  This package offers a generic support for different GNU Arm compilers.
    21     *  For the targets in this package, a user must set configuration parameter
    22     *  `LONGNAME` in `config.bld`. That parameter specifies the relative path to
    23     *  the compiler executable from the directory specified in `rootDir`.
    24     *
    25     *  @a(Example)
    26     *  If a version 2007q3 of the CodeSourcery GNU toolchain for Arm is installed
    27     *  in C:/CodeSourcery/arm-2007q3, the following settings in `config.bld`
    28     *  configure `gnu.targets.arm.GCArmv6` target to use that toolchain:
    29     *  @p(code)
    30     *  var GCArmv6 = xdc.module("gnu.targets.arm.GCArmv6");
    31     *  GCArmv6.rootDir = "C:/CodeSourcery/arm-2007q3";
    32     *  GCArmv6.LONGNAME = "bin/arm-none-linux-gnueabi-gcc";
    33     *  @p
    34     *
    35     *  @see gnu.targets.ITarget#LONGNAME
    36     *
    37     */
    38    package gnu.targets.arm [1,0,0,0] {
    39        module GCArmv5T, GCArmv6, GCArmv7A, M3, M4, M4F;
    40        module MVArm9, UCArm9;
    41    }
