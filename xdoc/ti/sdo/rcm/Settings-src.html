    44    /*!
    45     *  ======== Settings ========
    46     *  Global configuration settings for the ti.sdo.rcm package
    47     *
    48     *  This is a RTSC meta-only module. When building your executable with
    49     *  XDCtools, use this module to configure this package.
    50     *
    51     *  @a(See Also)
    52     *  @p(dlist)
    53     *  - {@link doxy(ti_sdo_rcm) RCM Overview}
    54     *  - {@link doxy(RcmClient.h) RcmClient File Reference}
    55     *  - {@link doxy(RcmServer.h) RcmServer File Reference}
    56     *  @p
    57     */
    58    
    59    metaonly module Settings
    60    {
    61    
    62        // -------- Module Constants --------
    63    
    64        // -------- Module Types --------
    65    
    66        /*!
    67         *  IPC Support enumeration type
    68         *
    69         *  The IPC support is provided by one of the following types.
    70         */
    71        enum IpcSupport {
    72            IpcSupport_ti_sdo_ipc,          //! DSP/BIOS IPC Support
    73            IpcSupport_ti_syslink_ipc       //! SysLink IPC Support
    74        };
    75    
    76    
    77        // -------- Module Parameters --------
    78    
    79        /*!
    80         *  Specifies which IPC support to link into the executable
    81         *
    82         *  This config param must be set in the application config script.
    83         *  It has no default value.
    84         */
    85        config IpcSupport ipc;
    86    
    87        /*!
    88         *  Controls the loading of string constants to the target
    89         *
    90         *  By default, all string constants are loaded to the target. If the
    91         *  program is using a logger which does not process the strings, then
    92         *  setting this config param to false will reduce the program's data
    93         *  footprint because the string constants will not be loaded.
    94         */
    95        config Bool loadStrings = true;
    96    }
