    37    /*! 
    38     *  ======== ILoggerMFP ========
    39     *  Extension of ILogger interface to allow more control of Log output.
    40     */
    41    interface ILoggerMFP inherits xdc.runtime.ILogger
    42    {
    43        /*!
    44         *  ======== Cmd ========
    45         *  Commands that can be sent to control function.
    46         */
    47        enum Cmd {
    48            SETTRACE = 0 /*! Set the trace output format based on the arg data
    49                          *  passed to the control() function.
    50                          */
    51        };
    52    
    53        /*!
    54         *  ======== control ========
    55         *  A hook for sending commands to the logger. For example, this can
    56         *  be used to re-configure the timestamp display format.
    57         *
    58         *  @param(cmd)         control command
    59         *  @param(cmdArgs)     command argument
    60         */
    61        Void control(Int cmd, UArg cmdArgs);
    62    }
