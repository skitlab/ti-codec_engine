    33    /*!
    34     *  ======== package.xdc ========
    35     *  Codec Engine SPHDEC-compliant codec
    36     *
    37     *  This codec illustrates the simplest integration into the Codec
    38     *  Engine.  Only three files are required to integrate into CE:
    39     *  @p(dlist)
    40     *      - `package.xdc`
    41     *          This file is required to declare a name for this package and to
    42     *          list all codec interfaces available.
    43     *      - `SPHDEC_COPY.xdc`
    44     *          This file specifies a codec (named SPHDEC_COPY) that is
    45     *          provided by this package.
    46     *      - `SPHDEC_COPY.xs`
    47     *          this file implements the functions required by the ICodec
    48     *          interface; these functions enable the codec to describe its
    49     *          non-functional requirements such as stack size.
    50     *
    51     *  All other files in this package are simply the implementation of the codec
    52     *  as specified by XDM's ISPHDEC interface.  Integration into CE
    53     *  requires no change to the codecs sources.
    54     *
    55     *  As the ISPHDEC interface is deprecated, the XDAIS product doesn't provide
    56     *  an example of an algorithm that implements the ISPHDEC interface.  This
    57     *  package serves two purposes:
    58     *  @p(blist)
    59     *      - Simple reference implementation of an algorithm that implements the
    60     *        ISPHDEC interface
    61     *      - Example of a package that both provides an algorithm's XDAIS/XDM
    62     *        libraries and implements the ti.sdo.ce.ICodec interface.
    63      *  @p
    64     */
    65    package ti.sdo.ce.examples.codecs.sphdec_copy [1, 0, 0] {
    66        module SPHDEC_COPY;
    67    }
