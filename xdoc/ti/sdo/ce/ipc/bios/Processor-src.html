    38    @Template("./Processor.xdt")

    40    metaonly module Processor {
    41        /*
    42         * sharedRegionIndex & clientHeapId - default values to use for all
    43         * Processor instances when no per-core values are provided in the
    44         * coreComm[] array.  The default values for numMsgs & msgSize in
    45         * CommDesc are taken from the existing Ipc.armCommNumMsgs &
    46         * Ipc.armCommMsgSize.
    47         */
    48        config UInt16 sharedRegionId = 0;
    49        config UInt16 heapId = 0;
    50    
    51        struct CommDesc {
    52            UInt32 numMsgs;
    53            UInt32 msgSize;
    54            UInt16 sharedRegionId;
    55            UInt16 heapId;
    56            Bool userCreatedHeap;
    57        };
    58        config CommDesc coreComm[];
    59    
    60        function close();
    61    }
