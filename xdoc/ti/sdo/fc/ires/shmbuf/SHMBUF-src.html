    34    /*!
    35     *  ======== SHMBUF ========
    36     *  Linux shared memory IRES Resman and Protocol Implementation
    37     */
    38    @Template("./SHMBUF.xdt")
    39    
    40    metaonly module SHMBUF {
    41    
    42        /*!
    43         *  ======== persistentAllocFxn ========
    44         *  Function for allocating memory for SHMBUF's internal objects.
    45         */
    46        config String persistentAllocFxn = null;
    47     
    48        /*!
    49         *  ======== persistentFreeFxn ========
    50         *  Function for freeing memory used by SHMBUF implementation's
    51         *  internal objects.
    52         *
    53         *  This is not required to be set.
    54         */
    55        config String persistentFreeFxn = null;
    56    
    57        /*!
    58         *  ======== key ========
    59         *  Key to identify buffer.
    60         */
    61        config UInt32 key = 0x53425546; /* "SBUF" */
    62    
    63        /*!
    64         *  ======== bufsize ========
    65         *  The length of the buffer resource.
    66         */
    67        config UInt32 bufsize = 0;
    68    
    69        /*!
    70         *  ======== isShared ========
    71         *  Can the buffer be assigned to multiple algorithms? If so, algorithms
    72         *  should not access the buffer simultaneously.
    73         */
    74        config bool isShared = true;
    75    
    76        config Bool enableLog;
    77    
    78        /*  @_nodoc */
    79        // Set to false until SHMBUFCFG_getIresmanFxns() and
    80        // SHMBUFCFG_getDefaultParams() are implemented.
    81        //override config Bool autoRegister = false;
    82    }
