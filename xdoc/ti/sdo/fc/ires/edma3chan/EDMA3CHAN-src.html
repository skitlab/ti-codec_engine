    38    
    39    metaonly module EDMA3CHAN inherits ti.sdo.fc.ires.IRESMAN {
    40    
    41        /*  @_nodoc */
    42        override config Bool autoRegister = true;
    43    
    44        /*!
    45         *  ======== ipcKeyBase ========
    46         *  Linux only. Base value of keys of Linux IPC objects used by
    47         *  EDMA3CHAN. The IPC objects created by EDMA3CHAN will use keys starting
    48         *  at this
    49         *  value, and incrementing with each new object. There are currently
    50         *  three IPC objects, so keys in the range of ipcKeyBase to ipcKeyBase + 2
    51         *  will be reserved for EDMA3CHAN. The default value of ipcKeyBase is
    52         *  ascii code for "NACH".
    53         *
    54         *  WARNING: This value should only be changed if it conflicts with
    55         *  another IPC key in the system that cannot be changed. If this value
    56         *  is changed, all programs using EDMA3CHAN that will be run simultaneously
    57         *  must have the ipcKeyBase configured to the new value.
    58         *  @_nodoc
    59         */
    60        config UInt ipcKeyBase = 0x4e414843;
    61    
    62    }
