    38    @Template("./TILEDMEMORY.xdt")

    41    metaonly module TILEDMEMORY inherits ti.sdo.fc.ires.IRESMAN {
    42    
    43        /*!
    44         *  ======= memoryBaseAddress =======
    45         *  
    46         */   
    47        config UInt tiledmemoryBaseAddress;
    48    
    49        /*!
    50         *  ======== ipcKeyBase ========
    51         *  Base value of keys of Linux IPC objects used by TILEDMEMORY.
    52         *
    53         *  This applies only to Linux-based environments. The IPC objects
    54         *  created by TILEDMEMORY will use keys starting at this
    55         *  value, and incrementing with each new object. There are currently
    56         *  three IPC objects, so keys in the range of ipcKeyBase to ipcKeyBase + 2
    57         *  will be reserved for TILEDMEMORY. The default value of ipcKeyBase is
    58         *  ascii code for "2VDH".
    59         *
    60         *  WARNING: This value should only be changed if it conflicts with
    61         *  another IPC key in the system that cannot be changed. If this value
    62         *  is changed, all programs using TILEDMEMORY that will be run 
    63         *  simultaneously must have the ipcKeyBase configured to the new value.
    64         *  @_nodoc
    65         */
    66        config UInt ipcKeyBase = 0x32564448;
    67    
    68        /*  @_nodoc */
    69        override config Bool autoRegister = true;
    70    }
