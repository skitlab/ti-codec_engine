    34    /*!
    35     *  ======== HDVICP ========
    36     *  HDVICP-specific IRES Resman and Protocol Implementation
    37     *
    38     */
    39    @Template("./HDVICP.xdt")
    40    
    41    metaonly module HDVICP inherits ti.sdo.fc.ires.IRESMAN {
    42    
    43        /*!
    44         *  ======== ipcKeyBase ========
    45         *  Base value of keys of Linux IPC objects used by HDVICP.
    46         *
    47         *  This applies only to Linux-based environments. The IPC objects
    48         *  created by HDVICP will use keys starting at this
    49         *  value, and incrementing with each new object. There are currently
    50         *  three IPC objects, so keys in the range of ipcKeyBase to ipcKeyBase + 2
    51         *  will be reserved for HDVICP. The default value of ipcKeyBase is
    52         *  ascii code for "IVDH".
    53         *
    54         *  WARNING: This value should only be changed if it conflicts with
    55         *  another IPC key in the system that cannot be changed. If this value
    56         *  is changed, all programs using HDVICP that will be run simultaneously
    57         *  must have the ipcKeyBase configured to the new value.
    58         *  @_nodoc
    59         */
    60        config UInt ipcKeyBase = 0x49564448;
    61    
    62        /*  @_nodoc */
    63        override config Bool autoRegister = true;
    64    }
